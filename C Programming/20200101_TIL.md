    ```(.C)
    void main()
    {
        char ch;
        ch = getchar();
        putchar(ch);
        ch = getchar();
        putchar(ch);
        ch = getchar();
        putchar(ch);
        // test 입력 시, 버퍼에 t,e,s,t,\n 값이 들어있으므로 새로운 입력 요청 없이 메모리에 보관된 test 값이 출력됨
    }
    ```
